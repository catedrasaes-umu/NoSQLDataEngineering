def getProps(EntityDiffSpec ent){
var commonPropsAux=ent.commonProps
  var List<Property> commonProps=new ArrayList
  for (i : 0 ..< commonPropsAux.size){
  	var ps=commonPropsAux.get(i)
  	var Property p=ps.property
  	if(p.class.getSimpleName()=="AggregateImpl"){
  	  var Aggregate agg=p as Aggregate
  	  var entAg=agg as Entity
  	  var eSp=findEntSpec(ps)
  	  
  	  commonAggrs.put(agg as Entity, (agg.name + (agg.eContainer.eContainer as Entity).name) + (agg.eContainer as EntityVersion).versionId )
   	}
 	commonProps.add(i,p)
  }
  
  commonAttrs=commonProps.filter(Attribute).toList
  //commonAggrs=commonProps.filter(Aggregate).toList
  commonRefs=commonProps.filter(Reference).toList
  var int contVer=-1
  for(entVer: ent.entityVersionProps){
    var eV=ent.entityVersionProps.get(contVer+=1)
    var notCommonPropsAux=eV.propertySpecs
    var List<Property> notCommonProps=new ArrayList
    for (i : 0 ..< notCommonPropsAux.size){
       var Property np=notCommonPropsAux.get(i).property
       var np2=np
  	   if(np2.class.getSimpleName()=="AggregateImpl"){
  	     var agg=np as Aggregate
  	     var ag2=agg as Entity
  	     var ev=agg.refTo.get(0)
  	     notCommonAggrs.put(ag2, (ag2.name + (agg.eContainer.eContainer as Entity).name) + (entVer.eContainer as EntityVersion).versionId)
  	   }
 	   notCommonProps.add(i,np)
    }
    var at=notCommonProps.filter(Attribute)
    var ref=notCommonProps.filter(Reference)
    var int contAt=-1
    var int contRf=-1
    var int contAgg=-1
    for (i:0..<at.size){
      notCommonAttrs.add(contAt+=1,at.get(i))
    }
    for (i:0..<ref.size){
      notCommonRefs.add(contRf+=1,ref.get(i))
    }
  }
}

 

def findEntSpec(PropertySpec ps){
  var ag=ps.property as Aggregate
  var ent=ag as Entity
  for(e:eDiffs){
	if(e.entity.name==ent.name)
	  return(e)
  }
}
